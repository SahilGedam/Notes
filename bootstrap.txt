6) cross component communication

==> parent to child

(parent html)

(parent ts)

(Child html)

<h1>hello {{item}}</h1>

==> child to parent

(parent html)

<app-child (updateDataEvent)="updateData($event)"></app-child>

(parent ts)

  updateData (data: any) {
    this.name=data;
  }

(Child html)

<input type="text" #box name="" id="">
<button (click)="updateDataEvent.emit(box.value)">update</button>

(child ts)

@Output() updateDataEvent=new EventEmitter <string>();
